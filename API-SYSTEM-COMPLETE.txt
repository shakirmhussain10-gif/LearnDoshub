REACT EXAMPLE:
═════════════════════════════════════════════════════════════════════════════

import { useState } from 'react';
import LearnDocsAPI from './api-endpoints.js';

export default function TextTool() {
  const [count, setCount] = useState(0);
  const api = new LearnDocsAPI();
  
  const handleCount = async (text) => {
    api.setAPIKey(process.env.REACT_APP_API_KEY);
    const result = await api.text.count(text);
    setCount(result.data);
  };
  
  return (
    <div>
      <textarea 
        placeholder="Paste text here..."
        onBlur={(e) => handleCount(e.target.value)} 
      />
      <p>Words: {count.words} | Chars: {count.characters}</p>
    </div>
  );
}


NODE.JS EXAMPLE:
═════════════════════════════════════════════════════════════════════════════

const LearnDocsAPI = require('./api-endpoints.js');
const api = new LearnDocsAPI();

async function main() {
  api.setAPIKey(process.env.LEARNDOCS_API_KEY);
  
  // Count words
  const count = await api.text.count('Hello World');
  console.log('Words:', count.data.words);
  
  // Convert case
  const upper = await api.text.convertCase('hello', 'upper');
  console.log('Uppercase:', upper.data.text);
  
  // Extract emails
  const emails = await api.text.extractEmails('Email: test@example.com');
  console.log('Emails:', emails.data.emails);
}

main();
