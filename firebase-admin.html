3. Deploy Firestore Rules

Copy these rules and deploy from Firebase CLI:

firestore.rules

rules_version = '2'; service cloud.firestore { match /databases/{database}/documents { match /settings/{settingId} { allow read: if true; allow write: if request.auth != null && request.auth.token.admin == true; } match /affiliate/{affiliateId} { allow read: if true; allow write: if request.auth != null && request.auth.token.admin == true; } match /feedback/{feedbackId} { allow read: if request.auth != null; allow create: if request.resource.data.keys().hasAll(['message', 'email']); allow write: if request.auth != null; } match /analytics/{docId} { allow read: if request.auth != null; allow create: if true; } match /admins/{adminId} { allow read: if request.auth != null; allow write: if request.auth != null && request.auth.token.admin == true; } } }
# Install Firebase CLI
npm install -g firebase-tools

# Login

# Deploy rules
firebase deploy --only firestore:rules
